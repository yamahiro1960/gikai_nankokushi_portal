台帳に紐づける

このログインアカウント（）を、台帳の人物に紐づけます。
紐づけると台帳の氏名・区分が反映され、ログイン状態を管理できます。

選択した台帳レコードのメールアドレスを上書き更新します。この操作は元に戻せません。

function normalizeEmail(value) { return (value || "").trim().toLowerCase(); } function escapeHtml(value) { return String(value || "") .replaceAll("&", "&") .replaceAll("<", "<") .replaceAll(">", ">") .replaceAll('"', """) .replaceAll("'", "'"); } function accessRoleToProfileRole(accessRole) { return accessRole === "管理者" ? "admin" : "viewer"; } function profileRoleToAccessRole(profileRole) { return profileRole === "admin" ? "管理者" : "使用者"; } function getLoginStatusBadge(isCurrent) { if (isCurrent === true) { return '有効'; } if (isCurrent === false) { return '無効'; } return '未連携'; } async function fetchProfiles(client) { const { data, error } = await client .from("profiles") .select("user_id,email,display_name,role") .order("email", { ascending: true }); if (error) { return { items: [], error }; } return { items: data || [], error: null }; } async function fetchMembers(client) { const full = await client .from("member_directory") .select("member_id,full_name,category,email,is_current,access_role") .order("member_id", { ascending: true }); if (!full.error) { return { items: full.data || [], error: null }; } const fallback = await client .from("member_directory") .select("member_id,full_name,category,email") .order("member_id", { ascending: true }); if (fallback.error) { return { items: [], error: fallback.error }; } return { items: (fallback.data || []).map((item) => ({ ...item, is_current: true, access_role: "使用者" })), error: null }; } function mergeRows(profileItems, memberItems) { const memberByEmail = new Map(); memberItems.forEach((member) => { const key = normalizeEmail(member.email); if (key) { memberByEmail.set(key, member); } }); const rows = profileItems.map((profile) => { const key = normalizeEmail(profile.email); const member = memberByEmail.get(key) || null; return { user_id: profile.user_id, member_id: member ? member.member_id : "", email: profile.email || (member ? member.email : ""), name: (member && member.full_name) || profile.display_name || "", category: member ? (member.category || "") : "", is_current: member ? !!member.is_current : null, role: profile.role || "viewer", hasProfile: true, hasMember: !!member }; }); const linkedEmails = new Set(rows.map((row) => normalizeEmail(row.email)).filter(Boolean)); memberItems.forEach((member) => { const key = normalizeEmail(member.email); if (!key || linkedEmails.has(key)) { return; } rows.push({ user_id: "", member_id: member.member_id || "", email: member.email || "", name: member.full_name || "", category: member.category || "", is_current: !!member.is_current, role: accessRoleToProfileRole(member.access_role), hasProfile: false, hasMember: true }); }); return rows.sort((a, b) => { return (a.email || "").localeCompare(b.email || "", "ja"); }); } function renderRows(items, client) { userRows.innerHTML = ""; const activeCount = items.filter((item) => item.is_current === true).length; const inactiveCount = items.filter((item) => item.is_current === false).length; const unlinkedCount = items.filter((item) => !item.hasMember).length; summary.textContent = `対象: ${items.length}件 / ログイン有効: ${activeCount}件 / 無効: ${inactiveCount}件 / 台帳未連携: ${unlinkedCount}件`; items.forEach((item) => { const roleOptions = item.hasProfile ? ["viewer", "editor", "admin"] : ["viewer", "admin"]; const roleOptionsHtml = roleOptions .map((roleValue) => ``) .join(""); const tr = document.createElement("tr"); tr.innerHTML = ` ${escapeHtml(item.email || "")}
${escapeHtml(item.name || "-")}
${escapeHtml(item.category || "未設定")}${item.hasMember ? "" : " / 台帳未連携"}
${getLoginStatusBadge(item.is_current)}
${escapeHtml(item.role || "viewer")} `; userRows.appendChild(tr); }); document.querySelectorAll(".saveRole").forEach((button) => { button.addEventListener("click", async () => { const userId = button.getAttribute("data-user-id"); const memberId = button.getAttribute("data-member-id"); const selector = `.roleSelect[data-user-id='${userId}'][data-member-id='${memberId}']`; const select = document.querySelector(selector); const role = select.value; if (!userId && role === "editor") { message.textContent = "profiles未連携のため、editor は設定できません。先に対象者を認証ユーザーとして作成してください。"; return; } if (userId) { const { error: profileError } = await client .from("profiles") .update({ role }) .eq("user_id", userId); if (profileError) { message.textContent = `権限更新失敗(profiles): ${profileError.message}`; return; } } if (memberId) { const accessRole = profileRoleToAccessRole(role); const { error: memberError } = await client .from("member_directory") .update({ access_role: accessRole }) .eq("member_id", memberId); if (memberError) { message.textContent = `権限更新失敗(member_directory): ${memberError.message}`; return; } } message.textContent = "権限を更新しました。"; await loadUsers(client); }); }); document.querySelectorAll(".toggleLogin").forEach((button) => { button.addEventListener("click", async () => { const memberId = button.getAttribute("data-member-id"); if (!memberId) { message.textContent = "台帳未連携のため、ログイン状態を変更できません。"; return; } const isCurrent = button.getAttribute("data-is-current") === "true"; const nextValue = !isCurrent; const { error } = await client .from("member_directory") .update({ is_current: nextValue }) .eq("member_id", memberId); if (error) { message.textContent = `ログイン状態更新失敗: ${error.message}`; return; } message.textContent = `ログイン状態を${nextValue ? "有効" : "無効"}に更新しました。`; await loadUsers(client); }); }); } async function loadUsers(client) { const profilesResult = await fetchProfiles(client); const membersResult = await fetchMembers(client); if (profilesResult.error && membersResult.error) { message.textContent = `読込失敗: ${profilesResult.error.message}`; return; } if (profilesResult.error && !membersResult.error) { message.textContent = `profiles 読込に失敗したため、台帳データのみ表示しています: ${profilesResult.error.message}`; } const mergedRows = mergeRows(profilesResult.items, membersResult.items); renderRows(mergedRows, client); } portalAuth.init({ requireAuth: true, minRole: "admin", authTargetId: "authArea", returnTo: "user-management.html", onReady: async ({ client }) => { await loadUsers(client); } });
